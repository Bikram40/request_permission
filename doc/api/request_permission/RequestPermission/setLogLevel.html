Future<void> setLogLevel(LogLevel logLevel) async {
  int levelCode = 8;

  switch (logLevel) {
    case LogLevel.verbose:
      levelCode = 2;
      break;
    case LogLevel.error:
      levelCode = 6;
      break;
    case LogLevel.none:
      levelCode = 8;
      break;
    default:
      levelCode = 8;
  }

  await _channel.invokeMethod(
    "setLogLevel",
    <String, int>{"logLevel": levelCode},
  );
}
